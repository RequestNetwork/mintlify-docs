## Advanced Subscription Features

<CodeGroup>
```javascript Subscription Creation
import { RequestNetwork } from '@requestnetwork/request-client.js';

async function createSubscription(customer, plan) {
  const subscription = await requestNetwork.createRequest({
    requestInfo: {
      currency: plan.currency,
      expectedAmount: plan.amount,
      payee: {
        type: 'ethereumAddress',
        value: merchantAddress
      },
      payer: {
        type: 'ethereumAddress', 
        value: customer.address
      }
    },
    contentData: {
      reason: `${plan.name} Subscription`,
      dueDate: plan.nextBillingDate,
      recurring: {
        frequency: plan.billingCycle, // 'monthly', 'yearly'
        occurrences: plan.maxOccurrences
      }
    }
  });
  
  return subscription;
}
```

```javascript Webhook Handling
app.post('/webhook/subscription', (req, res) => {
  const { eventType, request } = req.body;
  
  switch(eventType) {
    case 'recurring_payment_success':
      // Extend subscription period
      extendSubscription(request.customerId);
      break;
      
    case 'recurring_payment_failed':
      // Handle failed payment
      handleFailedPayment(request.customerId);
      break;
      
    case 'subscription_cancelled':
      // Process cancellation
      cancelSubscription(request.customerId);
      break;
  }
  
  res.status(200).send('OK');
});
```
</CodeGroup>
