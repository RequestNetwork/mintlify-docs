<ApiEndpoint method="POST" path="/v2/request" />

## Headers

<ParamField query="x-api-key" type="string" required>
  API key for authentication
</ParamField>

## Body

<ParamField body="payer" type="string">
  The wallet address of the payer
</ParamField>

<ParamField body="payee" type="string">
  The wallet address of the payee. Required for all requests except crypto-to-fiat
</ParamField>

<ParamField body="amount" type="string" required>
  The payable amount of the invoice, in human readable format
</ParamField>

<ParamField body="invoiceCurrency" type="string" required>
  Invoice Currency ID, from the [Request Network Token List](/general/request-network-token-list) e.g: USD
</ParamField>

<ParamField body="paymentCurrency" type="string" required>
  Payment currency ID, from the [Request Network Token List](/general/request-network-token-list) e.g: ETH-sepolia-sepolia
</ParamField>

<ParamField body="isCryptoToFiatAvailable" type="boolean">
  Whether crypto-to-fiat payment is available for this request
</ParamField>


<ParamField body="recurrence" type="object">
  The recurrence of the invoice
  
  <Expandable title="Recurrence Object Properties">
    <ParamField body="startDate" type="string | null" required>
      The start date of the invoice
    </ParamField>
    
    <ParamField body="frequency" type="string Â· enum" requred>
      The frequency of the invoice

      Possible values: `DAILY` `WEEKLY` `MONTHLY` `YEARLY`
    </ParamField>
  </Expandable>
</ParamField>

## Response

### 201 Request created successfully
<ResponseField name="Response" type="object">
<Expandable>
<ResponseField name="paymentReference" type="string" optional> 
Unique identifier of the request, used to pay the request as well as check the status of the request

Example:  `0xb3581f0b0f74cc61`
</ResponseField>

<ResponseField name="requestId" type="string" optional>
Unique identifier of the request, commonly used to look up a request in Request Scan

Example: `01e273ecc29d4b526df3a0f1f05ffc59372af8752c2b678096e49ac270416a7cdb`
</ResponseField>
</Expandable>
</ResponseField>
