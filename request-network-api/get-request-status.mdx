---
title: "Get Request Status"
description: "Get the status of a payment request"
api: GET https://api.request.network/v2/request/{requestId}
contentType: application/json
---




<ApiEndpoint method="GET" path="/v2/request/{requestId}" />

## Path Parameters

<ParamField path="requestId" type="string" required>
  The requestId for the request
  
  Example: `01e273ecc29d4b526df3a0f1f05ffc59372af8752c2b678096e49ac270416a7cdb`
</ParamField>

## Headers parameters

<ParamField query="x-api-key" type="string" required>
  API key for authentication
</ParamField>

## Response

### 200 Request status retrieved successfully
<ResponseField name="Response" type="object">
<Expandable>
<ResponseField name="hasBeenPaid" type="boolean" required>
Whether the request has been paid
</ResponseField>
<ResponseField name="paymentReference" type="string" required>
Unique identifier used for payments and status checks
</ResponseField>

<ResponseField name="requestId" type="string" required>
Unique identifier of the request
</ResponseField>

<ResponseField name="isListening" type="boolean" required>
Whether the system is actively listening for payments on this request
</ResponseField>

<ResponseField name="txHash" type="string | nullable" required>
Transaction hash of the payment, null if not yet paid
</ResponseField>

<ResponseField name="recurrence" type="object" optional>
Recurrence configuration for recurring requests
</ResponseField>

<ResponseField name="originalRequestId" type="string" optional>
Original request ID for recurring requests
</ResponseField>

<ResponseField name="status" type="string" optional>
Current status of the request
</ResponseField>

<ResponseField name="originalRequestPaymentReference" type="string" optional>
Payment reference of the original request for recurring payments
</ResponseField>

<ResponseField name="payments" type="object[]" optional>
Array of payments made to this request
</ResponseField>

<ResponseField name="isRecurrenceStopped" type="boolean" optional>
Whether recurrence has been stopped for this request
</ResponseField>

</Expandable>
</ResponseField>



## Error Responses

<ResponseField name="401" type="object">

</ResponseField>

<ResponseField name="429" type="object">
